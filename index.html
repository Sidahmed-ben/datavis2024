Requête pour récupérer des informations sur des pilotes de course : 

        #defaultView:Timeline
        SELECT ?painter ?date_of_birth ?place_of_birth ?place_of_birthLabel ?image ?coordinate_location ?painterLabel WHERE {
          SERVICE wikibase:label { bd:serviceParam wikibase:language "[AUTO_LANGUAGE],en". }
          ?painter wdt:P103 wd:Q150.
          OPTIONAL { ?painter wdt:P569 ?date_of_birth. }
          OPTIONAL { ?painter wdt:P19 ?place_of_birth. }
          ?painter wdt:P18 ?image.
          ?painter wdt:P641 wd:Q5386.
        }
        LIMIT 100
